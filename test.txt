import string
string.ascii_lowercase

def palindrome(s):
    #Remove the empty space if its a string
    s = s.replace(' ','')
    print(s)
    #Check is string is == reversed version of the string
    if  s == s[::-1]:
        print("Its Palindrome")
    else:
        print("Its not Palindrome")



#Calling the palindrome 

palindrome('madam')
palindrome('madam')
palindrome('nurses run')
